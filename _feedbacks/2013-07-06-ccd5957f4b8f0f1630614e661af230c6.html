in Anno_Template class:

public function get_linkedin_button($attr = array()) {
$default_attr = array(
'type' => "text/javascript",
'src' => "http://platform.linkedin.com/in.js",
);
$script = $this->utils->to_tag('script', ' ', $default_attr, $attr);
$default_attr = array(
'type' => "in/share",
'data-url' => get_permalink(),
);
return $script . $this->utils->to_tag('script', ' ', $default_attr, $attr);
}
